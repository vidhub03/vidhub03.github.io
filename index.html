--      List all shipments along with the name of their origin port. Include all shipments, 
--      even if the origin port does not exist in the Ports table. 
--      Show the shipment_id, origin_port_id, port_name, shipment_year, and shipment_weight_tonnes.
SELECT 
    s.shipment_id,
    s.origin_port_id,
    p.port_name AS origin_port_name
FROM dbo.Shipments s
LEFT JOIN dbo.Ports p
    ON s.origin_port_id = p.port_id;
